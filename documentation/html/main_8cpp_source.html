112void cursor_callback(GLFWwindow* window, double xdpos, double ydpos) {
113 float xpos = (float)xdpos;
114 float ypos = (float)ydpos;
115
116 if (firstMouse) {
117 lastX = xpos;
118 lastY = ypos;
119 firstMouse = false;
120 }
121
122 float xoffset = xpos - lastX;
123 float yoffset = lastY - ypos;
124 lastX = xpos;
125 lastY = ypos;
126
127 float sensitivity = 0.1f;
128 xoffset *= sensitivity;
129 yoffset *= sensitivity;
130
131 // mouse moving horizontally gives you yaw
132 // mouse moving vertically gives you pitch
133 yaw += xoffset;
134 pitch += yoffset;
135
136 if (pitch > 89.0f) pitch = 89.0f;
137 if (pitch < -89.0f) pitch = -89.0f;
138
139 glm::vec3 direction;
140
141 direction.x = cos(glm::radians(yaw)) * cos(glm::radians(pitch));
142 direction.y = sin(glm::radians(pitch));
143 direction.z = sin(glm::radians(yaw)) * cos(glm::radians(pitch));
144
145 cameraFront = glm::normalize(direction);
146 cameraRight = glm::normalize(glm::cross(cameraFront, worldUp));
147 cameraUp = glm::normalize(glm::cross(cameraRight, cameraFront));
148}
